**思路：n数据量很小，并且问题的描述其实是求一个最短哈密顿回路的长度，也就是旅行商问题，我们可以用暴力枚举来做其实，枚举10的全排列，也就是10!种可能性，然后用预处理的曼哈顿距离去算即可。当然我们也可以用状态压缩dp去进行优化处理，等会儿学学。**

**代码：**
```python
import sys
from itertools import permutations

input = lambda: sys.stdin.readline().strip()

import math

inf = 10 ** 18


def I():
    return input()


def II():
    return int(input())


def MII():
    return map(int, input().split())


def GMI():
    return map(lambda x: int(x) - 1, input().split())


def LI():
    return input().split()


def LII():
    return list(map(int, input().split()))


def LFI():
    return list(map(float, input().split()))


fmax = lambda x, y: x if x > y else y
fmin = lambda x, y: x if x < y else y
isqrt = lambda x: int(math.sqrt(x))

'''
旅行商问题，求解经过所有点的最短哈密顿回路，枚举10!种全排列，最终取fmin即可
'''

def solve():
    a, b = MII()
    sx, sy = MII()
    n = II()
    nums = [[sx, sy]] + [LII() for _ in range(n)]

    # 预处理曼哈顿距离
    d = [[0] * (n + 1) for _ in range(n + 1)]
    for i in range(n + 1):
        for j in range(n + 1):
            d[i][j] = abs(nums[i][0] - nums[j][0]) + abs(nums[i][1] - nums[j][1])
    
    # 枚举10!种可能，并且求出其中的最短距离
    ans = inf
    for perm in permutations(range(1, n + 1)):
        cur = d[0][perm[0]]
        for i in range(n - 1):
            cur += d[perm[i]][perm[i + 1]]
            if cur >= ans: # 优化，必须要有，不然超时
                break
        else: # 要算完了才能计入ans，所以说这里的写法是固定的
            ans = fmin(ans, cur + d[perm[-1]][0])
            
    print(f"The shortest path has length {ans}")


t = II()
for _ in range(t):
    solve()
```
